function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("votes"); const data = JSON.parse(e.postData.contents); sheet.appendRow([data.dilemma, data.choice, new Date()]); return ContentService.createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("votes"); const rows = sheet.getDataRange().getValues().slice(1); return ContentService.createTextOutput(JSON.stringify(rows)) .setMimeType(ContentService.MimeType.JSON); }
